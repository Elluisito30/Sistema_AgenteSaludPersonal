-- ============================================
-- HEALTH AI SYSTEM - DATABASE SCHEMA SIMPLE (PostgreSQL / Supabase)
-- ============================================

-- ============================================
-- 1. USERS (Usuarios)
-- ============================================
CREATE TABLE IF NOT EXISTS users (
    id SERIAL PRIMARY KEY,
    email VARCHAR(255) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    full_name VARCHAR(255),
    phone_number VARCHAR(20),
    created_at TIMESTAMP DEFAULT NOW(),
    last_login TIMESTAMP,
    is_active BOOLEAN DEFAULT TRUE
);

-- ============================================
-- 2. HEALTH PROFILES (Perfil de salud)
-- ============================================
CREATE TABLE IF NOT EXISTS health_profiles (
    id SERIAL PRIMARY KEY,
    user_id INTEGER REFERENCES users(id) ON DELETE CASCADE,
    age INTEGER,
    gender VARCHAR(10) CHECK (gender IN ('male', 'female')),
    height_cm DECIMAL(5,2),
    weight_kg DECIMAL(5,2),
    activity_level VARCHAR(20),
    sleep_hours INTEGER,
    smokes BOOLEAN DEFAULT FALSE,
    has_chronic_conditions BOOLEAN DEFAULT FALSE,
    chronic_conditions_detail TEXT,
    health_goals TEXT[], -- Array de objetivos
    chronic_diseases TEXT[], -- Enfermedades crónicas
    genetic_risk_factors TEXT[], -- Factores de riesgo genético
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW(),
    UNIQUE(user_id)
);

-- ============================================
-- 3. HEALTH ANALYSES (Análisis de salud)
-- ============================================
CREATE TABLE IF NOT EXISTS health_analyses (
    id SERIAL PRIMARY KEY,
    user_id INTEGER REFERENCES users(id) ON DELETE CASCADE,
    
    -- Métricas calculadas
    bmi DECIMAL(5,2),
    bmi_category VARCHAR(20),
    bmr DECIMAL(7,2),
    tdee DECIMAL(7,2),
    health_score DECIMAL(5,2),
    
    -- Evaluación de riesgo
    health_risk VARCHAR(20),
    fitness_level VARCHAR(20),
    
    -- Plan de salud (JSON)
    health_plan JSONB,
    nutrient_recommendations JSONB,
    predicted_improvements JSONB,
    
    -- Alertas y objetivos
    alerts JSONB,
    weekly_goals TEXT[],
    
    -- Fechas
    analyzed_at TIMESTAMP DEFAULT NOW(),
    next_checkup DATE,
    
    -- Confianza
    confidence_score DECIMAL(5,2)
);

CREATE INDEX IF NOT EXISTS idx_analyses_user ON health_analyses(user_id, id DESC);

-- ============================================
-- 4. DAILY PROGRESS (Progreso diario)
-- ============================================
CREATE TABLE IF NOT EXISTS daily_progress (
    id SERIAL PRIMARY KEY,
    user_id INTEGER REFERENCES users(id) ON DELETE CASCADE,
    date DATE NOT NULL,
    
    -- Métricas físicas
    weight_kg DECIMAL(5,2),
    steps_count INTEGER,
    exercise_minutes INTEGER,
    
    -- Nutrición
    calories_consumed DECIMAL(7,2),
    protein_g DECIMAL(6,2),
    carbs_g DECIMAL(6,2),
    fats_g DECIMAL(6,2),
    water_liters DECIMAL(4,2),
    
    -- Bienestar
    sleep_hours DECIMAL(3,1),
    sleep_quality INTEGER CHECK (sleep_quality BETWEEN 1 AND 10),
    stress_level INTEGER CHECK (stress_level BETWEEN 1 AND 10),
    mood INTEGER CHECK (mood BETWEEN 1 AND 10),
    energy_level INTEGER CHECK (energy_level BETWEEN 1 AND 10),
    
    -- Notas
    notes TEXT,
    
    created_at TIMESTAMP DEFAULT NOW(),
    UNIQUE(user_id, date)
);

CREATE INDEX IF NOT EXISTS idx_progress_user ON daily_progress(user_id, date DESC);

-- ============================================
-- 5. ML PREDICTIONS (Predicciones ML)
-- ============================================
CREATE TABLE IF NOT EXISTS ml_predictions (
    id SERIAL PRIMARY KEY,
    user_id INTEGER REFERENCES users(id) ON DELETE CASCADE,
    analysis_id INTEGER REFERENCES health_analyses(id) ON DELETE SET NULL,
    
    model_type VARCHAR(50) NOT NULL,
    model_version VARCHAR(20),
    
    prediction JSONB NOT NULL,
    confidence_score DECIMAL(5,4),
    
    input_features JSONB,
    
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX IF NOT EXISTS idx_predictions_user ON ml_predictions(user_id, id DESC);

-- ============================================
-- 6. ALERTS (Alertas)
-- ============================================
CREATE TABLE IF NOT EXISTS alerts (
    id SERIAL PRIMARY KEY,
    user_id INTEGER REFERENCES users(id) ON DELETE CASCADE,
    analysis_id INTEGER REFERENCES health_analyses(id) ON DELETE SET NULL,
    
    alert_type VARCHAR(50),
    priority VARCHAR(20),
    title VARCHAR(255),
    message TEXT,
    
    is_read BOOLEAN DEFAULT FALSE,
    read_at TIMESTAMP,
    
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX IF NOT EXISTS idx_alerts_user ON alerts(user_id, id DESC);

-- ============================================
-- 7. NOTIFICATIONS (Notificaciones enviadas)
-- ============================================
CREATE TABLE IF NOT EXISTS notifications (
    id SERIAL PRIMARY KEY,
    user_id INTEGER REFERENCES users(id) ON DELETE CASCADE,
    alert_id INTEGER REFERENCES alerts(id) ON DELETE SET NULL,
    
    channel VARCHAR(20), -- email, sms, whatsapp
    recipient VARCHAR(255),
    message TEXT,
    
    status VARCHAR(20), -- sent, failed, delivered
    sent_at TIMESTAMP,
    
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX IF NOT EXISTS idx_notifications_user ON notifications(user_id, id DESC);

-- ============================================
-- VISTAS ÚTILES
-- ============================================

-- Último análisis por usuario
CREATE OR REPLACE VIEW latest_analysis AS
SELECT DISTINCT ON (user_id)
    id, user_id, health_score, bmi, health_risk, analyzed_at
FROM health_analyses
ORDER BY user_id, id DESC;

-- Progreso del último mes
CREATE OR REPLACE VIEW last_month_progress AS
SELECT 
    user_id,
    date,
    weight_kg,
    exercise_minutes,
    sleep_hours,
    mood
FROM daily_progress
WHERE date >= CURRENT_DATE - INTERVAL '30 days'
ORDER BY user_id, date DESC;

-- Resumen por usuario
CREATE OR REPLACE VIEW user_summary AS
SELECT 
    u.id,
    u.email,
    u.full_name,
    hp.age,
    hp.weight_kg,
    hp.height_cm,
    la.health_score,
    la.health_risk,
    la.analyzed_at AS last_analysis_date,
    COUNT(DISTINCT ha.id) AS total_analyses
FROM users u
LEFT JOIN health_profiles hp ON u.id = hp.user_id
LEFT JOIN latest_analysis la ON u.id = la.user_id
LEFT JOIN health_analyses ha ON u.id = ha.user_id
GROUP BY 
    u.id, u.email, u.full_name, hp.age, hp.weight_kg, hp.height_cm, 
    la.health_score, la.health_risk, la.analyzed_at;
